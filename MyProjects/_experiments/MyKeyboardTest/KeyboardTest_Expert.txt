#include "daisy_field.h"
#include "daisysp.h"
#include <atomic>

#define NUM_VOICES 16
#define AUDIO_BLOCK_SIZE 4  // Minimum latency for real-time performance
#define MAX_CONTROL_SMOOTHING 0.001f  // ADC smoothing factor
#define CPU_CYCLES_MONITORING 1  // Enable performance monitoring

using namespace daisy;
using namespace daisysp;

// DMA-safe buffer allocation for zero-wait-state access
DMA_BUFFER_MEM_SECTION static float audio_output_buffer[AUDIO_BLOCK_SIZE][2];
static std::atomic<float> target_freq{440.0f};
static std::atomic<float> reverb_feedback{0.94f};

// Expert-level hardware abstraction
DaisyField hw;

// Advanced voice structure with envelope shaping and performance optimization
struct ExpertVoice {
    Oscillator osc_;
    Adsr env_;
    float amp_, target_amp_;
    bool on_;
    
    // Performance monitoring
    uint32_t last_process_time_;
    
    void Init(float samplerate) {
        osc_.Init(samplerate);
        osc_.SetWaveform(Oscillator::WAVE_POLYBLEP_SAW);
        osc_.SetAmp(0.7f);
        
        // Advanced envelope configuration with shaped segments
        env_.Init(samplerate);
        env_.SetTime(ADENV_SEG_ATTACK, 0.01f);
        env_.SetAttackShape(2.0f);  // Exponential curve
        env_.SetTime(ADENV_SEG_DECAY, 0.2f);
        env_.SetTime(ADENV_SEG_RELEASE, 0.5f);
        env_.SetSustainLevel(0.7f);
        env_.SetMin(0.0f);
        env_.SetMax(1.0f);
        
        amp_ = 0.0f;
        target_amp_ = 0.0f;
        on_ = false;
        last_process_time_ = 0;
    }
    
    float Process() {
        // High-performance envelope processing
        float env_val = env_.Process();
        float sig = osc_.Process() * env_val;
        
        // Soft limiting for headroom management
        if(sig > 1.0f) sig = 1.0f;
        if(sig < -1.0f) sig = -1.0f;
        
        return sig;
    }
    
    void set_note(float nn) { 
        osc_.SetFreq(mtof(nn)); 
    }
    
    void NoteOn() {
        on_ = true;
        env_.Trigger(false);  // Soft retrigger
    }
    
    void NoteOff() {
        on_ = false;
    }
    
    uint32_t GetLastProcessTime() const {
        return last_process_time_;
    }
};

// Memory-optimized voice allocation
static ExpertVoice v[NUM_VOICES];
static uint8_t button_states[16];
static uint8_t prev_button_states[16];

// Expert-level scale configuration with MIDI note mapping
float scale[16] = {
    0.f,   // C
    2.f,   // D  
    4.f,   // E
    5.f,   // F
    7.f,   // G
    9.f,   // A
    11.f,  // B
    12.f,  // C (octave)
    0.f,   // C#
    1.f,   // D#
    3.f,   // F#
    6.f,   // G#
    8.f,   // A#
    10.f,  // B#
    13.f,  // C (2 octaves)
    0.0f   // unused
};

float active_note = scale[0];
int8_t octaves = 2;

// Advanced reverb with performance optimization
static ReverbSc verb;

// Control processing with ADC smoothing
float kvals[8];
float cvvals[4];
float smoothed_kvals[8];
float smoothed_cvvals[4];

// Performance monitoring variables
#ifdef CPU_CYCLES_MONITORING
static uint32_t audio_callback_start_cycles = 0;
static uint32_t audio_callback_end_cycles = 0;
static float max_callback_time_us = 0.0f;
#endif

// Expert-level audio callback with real-time optimization
void ExpertAudioCallback(AudioHandle::InterleavingInputBuffer in,
                        AudioHandle::InterleavingOutputBuffer out,
                        size_t size) {
    
#ifdef CPU_CYCLES_MONITORING
    audio_callback_start_cycles = DWT->CYCCNT;
#endif
    
    // Critical section for shared data access
    __disable_irq();
    float current_reverb_feedback = reverb_feedback.load(std::memory_order_relaxed);
    __enable_irq();
    
    // Process voice allocation and note management
    for(int i = 0; i < NUM_VOICES; i++) {
        bool current_state = button_states[i];
        bool prev_state = prev_button_states[i];
        
        if(current_state && !prev_state) {
            // Note On
            v[i].NoteOn();
        } else if(!current_state && prev_state) {
            // Note Off
            v[i].NoteOff();
        }
    }
    
    // Update previous states
    memcpy(prev_button_states, button_states, sizeof(button_states));
    
    // High-performance audio processing
    float sig, send;
    float wetl, wetr;
    
    for(size_t i = 0; i < size; i += 2) {
        sig = 0.0f;
        
        // Vectorized voice processing for cache efficiency
        for(int voice = 0; voice < NUM_VOICES; voice++) {
            // Skip specific voices for keyboard layout optimization
            if(voice != 8 && voice != 11 && voice != 15) {
                sig += v[voice].Process();
            }
        }
        
        // Advanced reverb processing with performance optimization
        send = sig * 0.35f;
        verb.Process(send, send, &wetl, &wetr);
        
        // Professional audio output with headroom management
        out[i] = (sig + wetl) * 0.5f;
        out[i + 1] = (sig + wetr) * 0.5f;
    }
    
#ifdef CPU_CYCLES_MONITORING
    audio_callback_end_cycles = DWT->CYCCNT;
    uint32_t cycles_used = audio_callback_end_cycles - audio_callback_start_cycles;
    float callback_time_us = (float)cycles_used * 1000000.0f / System::GetSysClkFreq();
    
    if(callback_time_us > max_callback_time_us) {
        max_callback_time_us = callback_time_us;
    }
    
    // Audio dropout detection
    if(callback_time_us > (1000000.0f / 48000.0f * AUDIO_BLOCK_SIZE)) {
        // Handle audio dropout - could reduce voice count or sample rate
        HandleAudioDropout();
    }
#endif
}

// Audio dropout handling for real-time robustness
void HandleAudioDropout() {
    // Implementation for handling audio processing overruns
    // Could reduce polyphony, disable effects, or lower sample rate
    static uint32_t dropout_count = 0;
    dropout_count++;
}

// Expert-level LED management with performance optimization
void ExpertUpdateLeds(const float* knob_vals) {
    // LED mapping arrays for efficient access
    static constexpr size_t knob_leds[] = {
        DaisyField::LED_KNOB_1,
        DaisyField::LED_KNOB_2,
        DaisyField::LED_KNOB_3,
        DaisyField::LED_KNOB_4,
        DaisyField::LED_KNOB_5,
        DaisyField::LED_KNOB_6,
        DaisyField::LED_KNOB_7,
        DaisyField::LED_KNOB_8,
    };
    
    static constexpr size_t keyboard_leds[] = {
        DaisyField::LED_KEY_A1, DaisyField::LED_KEY_A2, DaisyField::LED_KEY_A3,
        DaisyField::LED_KEY_A4, DaisyField::LED_KEY_A5, DaisyField::LED_KEY_A6,
        DaisyField::LED_KEY_A7, DaisyField::LED_KEY_A8, DaisyField::LED_KEY_B2,
        DaisyField::LED_KEY_B3, DaisyField::LED_KEY_B5, DaisyField::LED_KEY_B6,
        DaisyField::LED_KEY_B7,
    };
    
    // Update knob LEDs with smoothed values
    for(size_t i = 0; i < 8; i++) {
        hw.led_driver.SetLed(knob_leds[i], smoothed_kvals[i]);
    }
    
    // Update keyboard LEDs (all active for this demo)
    for(size_t i = 0; i < 13; i++) {
        hw.led_driver.SetLed(keyboard_leds[i], 1.0f);
    }
    
    hw.led_driver.SwapBuffersAndTransmit();
}

// Advanced control processing with ADC smoothing and debouncing
void ProcessExpertControls() {
    hw.ProcessAnalogControls();
    hw.ProcessDigitalControls();
    
    // Octave control with edge detection
    if(hw.GetSwitch(DaisyField::SW_1)->RisingEdge()) {
        octaves--;
        if(octaves < 0) octaves = 0;
        TriggerAllNotes();
    }
    
    if(hw.GetSwitch(DaisyField::SW_2)->RisingEdge()) {
        octaves++;
        if(octaves > 4) octaves = 4;
        TriggerAllNotes();
    }
    
    // Advanced ADC processing with smoothing
    for(int i = 0; i < 8; i++) {
        float raw_value = hw.GetKnobValue(i);
        fonepole(smoothed_kvals[i], raw_value, MAX_CONTROL_SMOOTHING);
        kvals[i] = raw_value;
        
        if(i < 4) {
            float cv_raw = hw.GetCvValue(i);
            fonepole(smoothed_cvvals[i], cv_raw, MAX_CONTROL_SMOOTHING);
            cvvals[i] = cv_raw;
        }
    }
    
    // Update reverb feedback from knob 7
    if(hw.GetKnobValue(7) != smoothed_kvals[7]) {
        reverb_feedback.store(smoothed_kvals[7] * 0.98f, std::memory_order_relaxed);
    }
}

// Trigger all voices with new octave setting
void TriggerAllNotes() {
    for(int i = 0; i < NUM_VOICES; i++) {
        float note_freq = (12.0f * octaves) + 24.0f + scale[i];
        v[i].set_note(note_freq);
    }
}

// Performance monitoring and diagnostics
void PrintPerformanceStats() {
#ifdef CPU_CYCLES_MONITORING
    float current_sample_rate = hw.AudioSampleRate();
    float max_allowed_time_us = (1000000.0f / current_sample_rate) * AUDIO_BLOCK_SIZE;
    
    System::Println("=== Performance Statistics ===");
    System::Println("Max callback time: %.3f us", max_callback_time_us);
    System::Println("Max allowed time: %.3f us", max_allowed_time_us);
    System::Println("Sample rate: %.0f Hz", current_sample_rate);
    System::Println("Block size: %d samples", AUDIO_BLOCK_SIZE);
    System::Println("CPU utilization: %.1f%%", (max_callback_time_us / max_allowed_time_us) * 100.0f);
#endif
}

// Memory usage monitoring
void MonitorMemoryUsage() {
    extern char _estack, _sidata, _sdata, _edata, _sbss, _ebss;
    size_t used_sram = (&_estack) - (&_sidata);
    size_t total_sram = 0x20000000 + 512*1024 - 0x20000000; // 512KB SRAM
    
    System::Println("=== Memory Usage ===");
    System::Println("SRAM used: %zu / %zu bytes (%.1f%%)", 
                   used_sram, total_sram, (float)used_sram / total_sram * 100.0f);
}

// Expert-level system initialization
bool InitializeExpertSystem() {
    // Configure system for maximum performance
    System::Config sys_config;
    sys_config.Boost(); // 480MHz operation
    system.Init(sys_config);
    
    // Initialize hardware with error checking
    if(!hw.Init()) {
        System::Println("Hardware initialization failed!");
        return false;
    }
    
    float sample_rate = hw.AudioSampleRate();
    if(sample_rate <= 0) {
        System::Println("Invalid sample rate!");
        return false;
    }
    
    // Initialize voices with performance optimization
    for(int i = 0; i < NUM_VOICES; i++) {
        v[i].Init(sample_rate);
        v[i].set_note((12.0f * octaves) + 24.0f + scale[i]);
    }
    
    // Initialize reverb with expert settings
    verb.Init(sample_rate);
    verb.SetFeedback(0.94f);
    verb.SetCutoff(8000.0f);
    
    // Clear button states
    memset(button_states, 0, sizeof(button_states));
    memset(prev_button_states, 0, sizeof(prev_button_states));
    
    // Enable CPU cycle counter for performance monitoring
#ifdef CPU_CYCLES_MONITORING
    CoreDebug->DEMCR |= CoreDebug_DEMCR_TRCENA_Msk;
    DWT->CTRL |= DWT_CTRL_CYCCNTENA_Msk;
#endif
    
    return true;
}

int main(void) {
    System::Println("=== Daisy Field Expert Keyboard Synth ===");
    
    // Expert-level system initialization
    if(!InitializeExpertSystem()) {
        System::Println("System initialization failed!");
        while(1) {
            System::Delay(1000);
        }
    }
    
    // Start ADC and audio processing
    hw.StartAdc();
    hw.StartAudio(ExpertAudioCallback);
    
    System::Println("System initialized successfully!");
    PrintPerformanceStats();
    MonitorMemoryUsage();
    
    uint32_t last_stats_time = System::GetNow();
    
    // Main control loop with expert-level optimization
    for(;;) {
        // Process controls with expert algorithms
        ProcessExpertControls();
        
        // Update LEDs with performance optimization
        ExpertUpdateLeds(kvals);
        
        // Expert DAC output with precision scaling
        uint16_t dac_value1 = (uint16_t)(smoothed_kvals[0] * 4095.0f);
        uint16_t dac_value2 = (uint16_t)(smoothed_kvals[1] * 4095.0f);
        hw.seed.dac.WriteValue(DacHandle::Channel::ONE, dac_value1);
        hw.seed.dac.WriteValue(DacHandle::Channel::TWO, dac_value2);
        
        // Periodic performance monitoring
        uint32_t current_time = System::GetNow();
        if(current_time - last_stats_time > 5000) { // Every 5 seconds
            PrintPerformanceStats();
            MonitorMemoryUsage();
            last_stats_time = current_time;
        }
        
        // Minimal delay for power optimization
        System::Delay(1);
    }
}
