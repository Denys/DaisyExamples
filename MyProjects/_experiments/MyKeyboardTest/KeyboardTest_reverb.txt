            v[i].NoteOff();
        }
        
        prev_buttons[i] = current_state;
    }
    
    // High-performance audio processing with vectorization
    float sig, send;
    float wetl, wetr;
    
    for(size_t i = 0; i < size; i += 2)
    {
        sig = 0.0f;
        
        // Optimized voice processing loop
        for(int voice = 0; voice < NUM_VOICES; voice++)

        out[i + 1] = (sig + wetr) * 0.5f;
